32 /**
33 * @brief Function which generate error message based on pcre2 error code
34 * @ingroup pcre2cpp
35 * @tparam utf UTF type
36 * @param error_code pcre2 error code
37 * @return string value of type compatible with UTF type
38 */
39 template<utf_type utf>
40 static _PCRE2CPP_CONSTEXPR17 typename utils::pcre2_data<utf>::string_type generate_error_message(
41 const int error_code
42 ) noexcept {
43 using _pcre2_data_t = utils::pcre2_data<utf>;
44 using _string_type = typename _pcre2_data_t::string_type;
45 using _uchar_type = typename _pcre2_data_t::uchar_type;
46 using _char_type = typename _pcre2_data_t::string_char_type;
47
48 _uchar_type error_message[120];
49 if (const int size = _pcre2_data_t::get_error_message(error_code, error_message, 120); size != PCRE2_ERROR_BADDATA) {
50 return _string_type(reinterpret_cast<_char_type*>(error_message), 120);
51 }
52 return _string_type();
53 }

55 /**
56 * @brief Function which generates error message based on pcre2 error code and with additional information about offset
57 * @ingroup pcre2cpp
58 * @tparam utf UTF type
59 * @param error_code pcre2 error code
60 * @param error_offset offset for example in regex pattern
61 * @return string value of type compatible with UTF type
62 */
63 template<utf_type utf>
64 static _PCRE2CPP_CONSTEXPR17 typename utils::pcre2_data<utf>::string_type generate_error_message(const int error_code,
65 const size_t error_offset) noexcept {
66 using _string_type = typename utils::pcre2_data<utf>::string_type;
67
68 #if _PCRE2CPP_HAS_UTF8
69 if _PCRE2CPP_CONSTEXPR17 (utf == utf_type::UTF_8) {
70 return fmt::format("error at {} {}", error_offset, pcre2cpp::generate_error_message<utf>(error_code));
71 }
72 else
73 #endif
74 #if _PCRE2CPP_HAS_UTF16
75 if _PCRE2CPP_CONSTEXPR17 (utf == utf_type::UTF_16) {
76 return fmt::format(u"error at {} {}", error_offset, pcre2cpp::generate_error_message<utf>(error_code));
77 }
78 else
79 #endif
80 #if _PCRE2CPP_HAS_UTF32
81 if _PCRE2CPP_CONSTEXPR17 (utf == utf_type::UTF_32) {
82 return fmt::format(U"error at {} {}", error_offset, pcre2cpp::generate_error_message<utf>(error_code));
83 }
84 else
85 #endif
86 {
87 return _string_type();
88 }
89 }

91 /**
92 * @brief converts any message from any utf to utf-8
93 * @ingroup pcre2cpp
94 * @tparam utf UTF type
95 * @param message message to convert
96 * @return std::string value of message
97 */
98 template<utf_type utf>
99 static _PCRE2CPP_CONSTEXPR20 std::string convert_any_utf_to_utf8(const typename utils::pcre2_data<utf>::string_view_type message) noexcept {
100 #if _PCRE2CPP_HAS_UTF8
101 if _PCRE2CPP_CONSTEXPR17 (utf == utf_type::UTF_8) { return std::string(message); }
102 else
103 #endif
104 #if _PCRE2CPP_HAS_UTF16
105 if _PCRE2CPP_CONSTEXPR17 (utf == utf_type::UTF_16) {
106 std::string msg;
107 for (const auto& c : message) { msg += static_cast<std::string::value_type>(c); }
108 return msg;
109 }
110 else
111 #endif
112 #if _PCRE2CPP_HAS_UTF32
113 if _PCRE2CPP_CONSTEXPR17 (utf == utf_type::UTF_32) {
114 std::string msg;
115 for (const auto& c : message) { msg += static_cast<std::string::value_type>(c); }
116 return msg;
117 }
118 else
119 #endif
120 {
121 return std::string();
122 }
123 }

133 template<utf_type utf>
134 class basic_pcre2cpp_exception : public std::runtime_error {
135 private:
136 using _pcre2_data_t = utils::pcre2_data<utf>;
137 using _string_type = typename _pcre2_data_t::string_type;
138 using _string_view_type = typename _pcre2_data_t::string_view_type;
139 using _uchar_type = typename _pcre2_data_t::uchar_type;
140 using _char_type = typename _pcre2_data_t::string_char_type;
141
142 /// @brief error message
143 _string_type _message;
144
145 public:
146 /// @brief constructor with message
147 explicit basic_pcre2cpp_exception(const _string_view_type message) noexcept
148 : std::runtime_error(convert_any_utf_to_utf8<utf>(message)), _message(message) {}
149
150 /// @brief constructor with error code
151 explicit basic_pcre2cpp_exception(const int error_code) noexcept
152 : std::runtime_error(convert_any_utf_to_utf8<utf>(generate_error_message<utf>(error_code))),
153 _message(generate_error_message<utf>(error_code)) {}
154
155 /// @brief constructor with error code and error offset
156 basic_pcre2cpp_exception(const int error_code, const size_t error_offset) noexcept
157 : std::runtime_error(convert_any_utf_to_utf8<utf>(generate_error_message<utf>(error_code, error_offset))),
158 _message(generate_error_message<utf>(error_code, error_offset)) {}
159
160 /// @brief returns error message
161 _PCRE2CPP_CONSTEXPR17 const _string_type& get_error() const noexcept { return _message; }
162 };
