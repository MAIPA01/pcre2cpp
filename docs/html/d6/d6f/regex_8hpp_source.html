62 _code_ptr _code = nullptr;
63 /// @brief pointer to match data of pcre2 code
64 _match_data_ptr _match_data = nullptr;
65 /// @brief pointer to conversion table of named groups to their index
66 _named_sub_values_table_ptr _named_sub_values = nullptr;

68 public:
69 /// @brief basic regex container with pattern and compile options
70 _PCRE2CPP_CONSTEXPR20 explicit basic_regex(const _string_view_type pattern,
71 const compile_options opts = compile_options_bits::None) _PCRE2CPP_NOEXCEPT {
72 // Compile Code
73 int error_code;
74 size_t error_offset;
75
76 _code_type* code = _pcre2_data_t::compile(reinterpret_cast<_sptr_type>(pattern.data()), pattern.size(), opts,
77 &error_code, &error_offset, nullptr);
78
79
80 #if !_PCRE2CPP_HAS_EXCEPTIONS
81 pcre2cpp_assert(code != nullptr, "Failed to initialize code: {}",
82 convert_any_utf_to_utf8<utf>(generate_error_message<utf>(error_code, error_offset)));
83 #else
84 if (code == nullptr) { throw _regex_exception(error_code, error_offset); }
85 #endif
86
87
88 _code = std::shared_ptr<_code_type>(code, _pcre2_data_t::code_free);
89
90 // Get Named Sub Values
91 _named_sub_values = std::make_shared<_named_sub_values_table>();
92
93 size_t name_count = 0;
94 _uchar_type* name_table = nullptr;
95 size_t name_entry_size = 0;
96
97 _pcre2_data_t::get_info(_code.get(), PCRE2_INFO_NAMECOUNT, &name_count);
98 _pcre2_data_t::get_info(_code.get(), PCRE2_INFO_NAMETABLE, &name_table);
99 _pcre2_data_t::get_info(_code.get(), PCRE2_INFO_NAMEENTRYSIZE, &name_entry_size);
100
101 for (size_t i = 0; i != name_count; ++i) {
102 _uchar_type* entry = name_table + i * name_entry_size + 2;
103 const int index = _pcre2_data_t::substring_number_from_name(_code.get(), entry);
104
105 _uchar_type* entry_end = entry + 1;
106 while (*entry_end != 0 && entry_end - entry < name_entry_size - 3) { entry_end += 1; }
107 _named_sub_values->emplace(_string_view_type(reinterpret_cast<_string_char_type*>(entry), entry_end - entry),
108 static_cast<size_t>(index) - 1);
109 }
110
111 // Create Match Data
112 _match_data_type* match_data = _pcre2_data_t::match_data_from_pattern(_code.get(), nullptr);
113 _match_data = std::shared_ptr<_match_data_type>(match_data, _pcre2_data_t::match_data_free);
114 }

129 /// @brief returns true if match was found
130 _PCRE2CPP_CONSTEXPR17 bool match(const _string_view_type text, const size_t offset = 0,
131 const match_options opts = match_options_bits::None) const _PCRE2CPP_NOEXCEPT {
132 const int match_code = _pcre2_data_t::match(_code.get(), reinterpret_cast<_sptr_type>(text.data()), text.size(),
133 offset, opts, _match_data.get(), nullptr);
134
135 return match_code != static_cast<int>(match_error_codes::NoMatch) && match_code > 0;
136 }

138 /// @brief returns true if match was found and result is stored in result variable
139 _PCRE2CPP_CONSTEXPR20 bool match(const _string_view_type text, _match_result_type& result, const size_t offset = 0,
140 const match_options opts = match_options_bits::None) const noexcept {
141 const int match_code = _pcre2_data_t::match(_code.get(), reinterpret_cast<_sptr_type>(text.data()), text.size(),
142 offset, opts, _match_data.get(), nullptr);
143
144 if (match_code == static_cast<int>(match_error_codes::NoMatch) || match_code <= 0) {
145 result = _match_result_type(static_cast<match_error_codes>(match_code));
146 return false;
147 }
148
149 const size_t* offsetVector = _pcre2_data_t::get_ovector_ptr(_match_data.get());
150 const size_t matchStart = offsetVector[0];
151 const size_t matchEnd = offsetVector[1];
152 _match_value_type matchValue = {
153 .relative_offset = matchStart - offset,
154 .value = _string_type(text.substr(matchStart, matchEnd - matchStart)),
155 };
156
157 const size_t offsetVectorsCount = _pcre2_data_t::get_ovector_count(_match_data.get());
158
159 std::vector<std::optional<sub_match_value> > subMatchValues;
160 subMatchValues.reserve(offsetVectorsCount);
161 for (size_t i = 1; i != offsetVectorsCount; ++i) {
162 const size_t subMatchStart = offsetVector[i * 2];
163 const size_t subMatchEnd = offsetVector[i * 2 + 1];
164
165 if (subMatchStart == PCRE2_UNSET || subMatchEnd == PCRE2_UNSET) { subMatchValues.emplace_back(); }
166 else {
167 subMatchValues.push_back(sub_match_value {
168 .relative_offset = subMatchStart - matchStart,
169 .size = subMatchEnd - subMatchStart,
170 });
171 }
172 }
173
174 result = _match_result_type(offset, matchValue, subMatchValues, _named_sub_values, _code);
175 return true;
176 }

178 /// @brief returns true if match was found, and it has relative offset == 0
179 _PCRE2CPP_CONSTEXPR17 bool match_at(const _string_view_type text, const size_t offset = 0) const noexcept {
180 _match_result_type result;
181 return match_at(text, result, offset);
182 }

184 /// @brief returns true if match was found, and it has relative offset == 0 and result is stored in result variable
185 _PCRE2CPP_CONSTEXPR17 bool match_at(const _string_view_type text, _match_result_type& result,
186 const size_t offset = 0) const noexcept {
187 if (!match(text, result, offset)) { return false; }
188
189 if (result.get_result_relative_offset() != 0) {
190 result = _match_result_type(offset, _named_sub_values, _code);
191 return false;
192 }
193
194 return true;
195 }
196
197 /// @brief returns true if any match was found and all results store in results array
198 _PCRE2CPP_CONSTEXPR17 bool match_all(const _string_view_type text, std::vector<_match_result_type>& results,
199 size_t offset = 0) const noexcept {
200 size_t start_offset = offset;
201 _match_result_type result;
202 while (match(text, result, offset)) {
203 results.emplace_back(start_offset,
204 _match_value_type {
205 .relative_offset = offset - start_offset + result.get_result_relative_offset(),
206 .value = result.get_result_value(),
207 },
208 result.get_sub_results(), _named_sub_values, _code);
209
210 offset += result.get_result_relative_offset() + result.get_result_size();
211 }
212
213 return results.size() != 0;
214 }
